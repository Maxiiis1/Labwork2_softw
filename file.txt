
#include <iostream>
#include <vector>

using namespace std;

int main() {
	int n;
	cin >> n;
	vector <int> a(n + 1);
	int ch, pos;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	cin >> ch >> pos;
	for (int i = n - 1; i >= pos - 1; i--) {
		a[i + 1] = a[i];
	} 
	a[pos - 1] = ch;
	for (int i = 0; i <= n; i++)
		cout << a[i] << " ";
}
